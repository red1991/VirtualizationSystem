 71 /**
 72  * Funzione di accensione della macchina virtuale.
 73  *
 74  * @param {int} idVm - Identificativo univoco della macchina virtuale.
 75  */
 76 function runVm(idVm) {
 77 $.ajax({
 78 type: "POST",
 79 url: "https://192.168.1.70/runVm.php",
 80 data: "idVm=" + idVm,
 81 dataType: "html",
 82       
 83 success: function(response)
 84 {
 85 var dec = JSON.parse(response);
 86 if(dec.status == 'failed'){
 87 if(dec.fault.detail == '[Cannot run VM because the VM is in Wait for Launch status.]'){
 88 document.getElementById("textLog").style.cssText = "color:red;"
 89 document.getElementById("textLog").innerHTML = "Macchina già accesa!";
 90 }
 91 else if(dec.fault.detail == "[Cannot run VM because the VM is in Powering Up status.]") {
 92 document.getElementById("textLog").style.cssText = "color:red;"
 93 document.getElementById("textLog").innerHTML = "Macchina già accesa!";
 94 }
 95 else if(dec.fault.detail == "[Cannot run VM because the VM is in Up status.]") {
 96 document.getElementById("textLog").style.cssText = "color:red;"
 97 document.getElementById("textLog").innerHTML = "Macchina già accesa!";
 98 }
 99 else if(dec.fault.detail == "[Cannot run VM. Unknown Data Center status.]") {
100 document.getElementById("textLog").style.cssText = "color:red;"
101 document.getElementById("textLog").innerHTML = "Data Center non riconosciuto!";
102 } 
103 else {
104 document.getElementById("textLog").style.cssText = "color:red;"
105 document.getElementById("textLog").innerHTML = dec.fault.detail;
106 } 
107 }
108        
109 if(dec.status == 'complete'){
110 logRunInit = "<-- Avvio macchina virtuale -->";
111 document.getElementById("textLog").style.cssText = "color:black;"
112 document.getElementById("textLog").innerHTML = logRunInit;
113 var createAccensioneTimer = setInterval(function() { createAccensione(idVm,logRunInit,createAccensioneTimer); }, 8000);
114 }    	
115 },
116 error: function(){}
117 }); 
118 }

297 /**
298  * Funzione di controllo stato macchina virtuale.
299  *
300  * @param {int} idVm - Identificativo univoco della macchina virtuale.
301  */
302 function checkStato(idVm) {
303 $.ajax({
304 type: "POST",
305 url: "https://192.168.1.70/getVmApi.php",
306 data: "idVm=" + idVm,
307 dataType: "html",
308       
309 success: function(response)
310 {
311 var dec = JSON.parse(response);  	   
312 if(dec.status == "wait_for_launch" || dec.status == "powering_up"){
313 document.getElementById("submitRun").disabled = true;
314 document.getElementById("submitShut").disabled = true;
315 document.getElementById("submitConsole").disabled = true;
316 document.getElementById("submitDelete").disabled = true;
317 document.getElementById("submitDeploy").disabled = true;
318 }
319 if(dec.status == "up"){
320 document.getElementById("submitConsole").disabled = false;
321 document.getElementById("submitShut").disabled = false;
322 document.getElementById("submitDelete").disabled = true;
323 document.getElementById("submitRun").disabled = true;
324 document.getElementById("submitDeploy").disabled = true;
325 }
326 else if(dec.status == "down") {
327 document.getElementById("submitShut").disabled = true;
328 document.getElementById("submitRun").disabled = false;
329 document.getElementById("submitConsole").disabled = true;
330 document.getElementById("submitDelete").disabled = false;
331 }
332 else if(dec.status == "powering_down") {
333 document.getElementById("submitRun").disabled = true;
334 document.getElementById("submitShut").disabled = false;
335 document.getElementById("submitConsole").disabled = true;
336 document.getElementById("submitDelete").disabled = true;
337 document.getElementById("submitDeploy").disabled = true;
338 }
339 else if(dec.status == "image_locked") {
340 document.getElementById("submitRun").disabled = true;
341 document.getElementById("submitShut").disabled = true;
342 document.getElementById("submitConsole").disabled = true;
343 document.getElementById("submitDelete").disabled = true;
344 document.getElementById("submitDeploy").disabled = true;
345 }
346 },
347 error: function(){}
348 }); 	
349 }
