 11 function selectProgrammi(){
 12 var templateSelezionato = document.formAddVm.selectTemplate.selectedIndex;
 13 var campo = document.formAddVm.selectTemplate.options;
 14 var template = campo[templateSelezionato].text;
 15 if(template == "CentOS-7-x86_64-DVD-1804"){
 16 document.getElementById("checkCB").checked = false;
 17 document.getElementById("checkLO").checked = false;
 18 document.getElementById("checkCB").disabled = false;
 19 document.getElementById("checkLO").disabled = false;
 20 document.getElementById("textNomeHost").disabled = false;
 21 }
 22 else if (template == "Ubuntu-16.04.4-desktop-amd64") {
 23 document.getElementById("checkCB").disabled = true;
 24 document.getElementById("checkLO").disabled = true;
 25 document.getElementById("checkCB").disabled = true;
 26 document.getElementById("checkLO").disabled = true;
 27 document.getElementById("textNomeHost").disabled = true;
 28 }
 29 }

 31 $.ajax({
 32 type: "GET",
 33 url: "https://192.168.1.70/getTemplatesApi.php",             
 34 success: function(response){
 35 var dec = JSON.parse(response); 			
 36 var key, count = 0;
 37 for(key in dec.template) {
 38 if(dec.template.hasOwnProperty(key)) {
 39 count++;
 40 }
 41 }
 42 for(var i = 0; i < count; i++){
 43 var selectTemplate = document.formAddVm.selectTemplate;
 44 if (dec.template[i].name == "CentOS-7-x86_64-DVD-1804" || dec.template[i].name == "Ubuntu-16.04.4-desktop-amd64") {
 45 var option = document.createElement("option");
 46 option.text = dec.template[i].name;
 47 option.value = i;
 48 selectTemplate.add(option);
 49 }				
 50 }
 51 },
 52 error: function(){
 53 document.getElementById("textLog").style.cssText = "color:red;"
 54 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
 55 }

 58 $("#submitCreate").click(function(){	
 59 var templateSelezionato = document.formAddVm.selectTemplate.selectedIndex;
 60 var campo = document.formAddVm.selectTemplate.options;
 61 var template = campo[templateSelezionato].text;
 62 var nome = $("#textNome").val();
 63 var descrizione = $("#textDescrizione").val();
 64 var dimensioneRam = $("#textDimensioneRam").val();
 65 var nomeHost = $("#textNomeHost").val();
 66 var checkedCB = document.getElementById("checkCB").checked;
 67 var checkedLO = document.getElementById("checkLO").checked; 
 68 
 69 /**
 70  * Crea un oggetto MV dai parametri passati.
 71  *
 72  * @param {string} nome - Nome della macchina virtuale.
 73  * @param {string} descrizione - Descrizione della macchina virtuale.
 74  * @param {int} dimensioneRam - Dimensione della memoria RAM.
 75  * @param {string} nomeHost - Hostname della macchina virtuale.
 76  * @param {boolean} checkedCB - Checkbox CodeBlocks.
 77  * @param {boolean} checkedLO - Checkbox LibreOffice.
 78  */
 79 var mv = new MV(nome, descrizione, dimensioneRam, nomeHost, checkedCB, checkedLO);
 80 
 81 /**
 82  * Crea un oggetto Template a partire dai parametri passati.
 83  *
 84  * @param {string} template - Nome del template.
 85  */
 86 var tmp = new Template(template);
 87 
 88 var errore = "";
 89 var send = true;
 90 
 91 if(mv.nome == "" || mv.descrizione == "" || mv.dimensioneRam == ""){
 92 document.getElementById("textLog").style.cssText = "color:red;"
 93 errore = errore + " Ci sono campi vuoti!<br>";
 94 send = false;
 95 }
 96 
 97 if(!(mv.dimensioneRam >= 2 && mv.dimensioneRam <= 4)){
 98 document.getElementById("textLog").style.cssText = "color:red;"
 99 errore = errore + " Dimensione ram tra 2 e 4 GB!<br>";
100 send = false;
101 }
102 
103 if(isNaN(mv.dimensioneRam)){
104 document.getElementById("textLog").style.cssText = "color:red;"
105 errore = errore + " Dimensione ram non numerica!<br>";
106 send = false;
107 }
108 
109 document.getElementById("textLog").innerHTML = errore;
110 
111 if(send){
112 $.ajax({
113 type: "POST",
114 url: "https://192.168.1.70/addVm.php",
115 data: "nome=" + mv.nome + "&descrizione=" + mv.descrizione + "&dimensioneRam=" + mv.dimensioneRam + "&template=" + tmp.template,
116 dataType: "html",  
117 success: function(response)
118 {
119 var log = "";
120 var dec = JSON.parse(response);
121 var stato = dec.detail;
122 if(stato == "[Cannot add VM. The VM name is already in use, please choose a unique name and try again.]"){
123 document.getElementById("textLog").style.cssText = "color:red;"
124 document.getElementById("textLog").innerHTML = "Nome già in uso!";
125 }
126 else if(stato == "[Cannot add VM. Unknown Data Center status.]") {
127 document.getElementById("textLog").style.cssText = "color:red;"
128 document.getElementById("textLog").innerHTML = "Data Center non riconosciuto!";
129 }
130 else if(stato == "[Can not add VM. The given name contains special characters. Only lower-case and upper-case letters, numbers, '_', '-', '.' are allowed., Attribute: vmStatic.name]") {
131 document.getElementById("textLog").style.cssText = "color:red;"
132 document.getElementById("textLog").innerHTML = "Caratteri non consentiti! ";
133 }
134 else if(stato == "For correct usage, see: https://192.168.1.70/ovirt-engine/apidoc#services/vms/methods/add") {
135 document.getElementById("textLog").style.cssText = "color:red;"
136 document.getElementById("textLog").innerHTML = "Si è verificato un errore. Riprovare!";
137 }
138 else if(stato == undefined) {
139 document.getElementById("submitCreate").disabled = true;
140 logInit = "<-- Inizializzazione -->";
141 document.getElementById("textLog").style.cssText = "color:black;"
142 document.getElementById("textLog").innerHTML = logInit; 
143 var createVmTimer = setInterval(function() { createVm(dec,logInit, createVmTimer); }, 3000); 		
144 }
145 else {
146 document.getElementById("textLog").style.cssText = "color:red;"
147 document.getElementById("textLog").innerHTML = stato;
148 }    	   	
149 },
150 error: function()
151 {
152 document.getElementById("textLog").style.cssText = "color:red;"
153 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
154 }
155 });   
156 }

158 /*
159  * Funzione che controlla il processo di creazione di una nuova macchina virtuale.
160  *
161  * @param {Json} dec - Json contenente la risposta di oVirt.
162  * @param {string} logInit - Messaggio di log.
163  * @param {double} createVmTimer - Timer elaborazione.
164  */
165 function createVm(dec,logInit, createVmTimer){
166 var logCreate = "Creazione macchina virtuale in corso..." + "<br>" + logInit;
167 document.getElementById("textLog").style.cssText = "color:black;"
168 document.getElementById("textLog").innerHTML = logCreate;                
169 var idVm = dec.id;  
170      
171 $.ajax({
172 type: "POST",
173 url: "https://192.168.1.70/getVmApi.php",
174 data: "idVm=" + idVm,
175 dataType: "html",
176 success: function(response)
177 {
178 var dec = JSON.parse(response);	   
179 if(dec.status == "down"){
180 var logCreateSuccess = "Macchina creata con successo." + "<br>" + logCreate;
181 document.getElementById("textLog").style.cssText = "color:black;"
182 document.getElementById("textLog").innerHTML = logCreateSuccess;
183 clearInterval(createVmTimer);
184 if (template == "Ubuntu-16.04.4-desktop-amd64"){
185 setTimeout(function(){ completeCreateVm(dec,logCreateSuccess); }, 10000); 
186 }
187 else {
188 setTimeout(function(){ runInit(dec,logCreateSuccess); }, 10000); 
189 } 
190 }
191 },
192 error: function()
193 {
194 document.getElementById("textLog").style.cssText = "color:black;"
195 document.getElementById("textLog").style.cssText = "color:red;"
196 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
197 }
198 }); 
199 }
