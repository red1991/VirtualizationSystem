 78 /**
 79  * Funzione che riceve i dati inseriti dall'utente contenenti le caratteristiche del pool da creare.
 80  *
 81  * @param {int} idVm - Identificativo univoco della macchina virtuale
 82  */
 83 function deploy(idVm) {
 84 var textNome = $("#textNome").val();
 85 var textDescrizione = $("#textDescrizione").val();
 86 var textNumero = $("#textNumero").val();
 87 
 88 /**
 89  * Crea un oggetto Pool a partire dai parametri passati.
 90  *
 91  * @param {string} textNome - Nome del pool.
 92  * @param {string} textDescrizione - Descrizione del pool.
 93  * @param {int} textNumero - Numero di macchine virtuali del pool.
 94  */
 95 var pool = new Pool(textNome, textDescrizione, textNumero);
 96 
 97 var errore = "";
 98 var send = true;
 99 
100 if(pool.textNome == "" || pool.textDescrizione == "" || pool.textNumero == ""){
101 document.getElementById("textLog").style.cssText = "color:red;"
102 errore = errore + " Ci sono campi vuoti!<br>";
103 send = false;
104 }
105 
106 if(isNaN(pool.textNumero)){
107 document.getElementById("textLog").style.cssText = "color:red;"
108 errore = errore + " Campo non numerico!<br>";
109 send = false;
110 }
111 
112 if(!(pool.textNumero >= 2 && pool.textNumero <= 5)){
113 document.getElementById("textLog").style.cssText = "color:red;"
114 errore = errore + " Numero macchine tra 2 e 5! <br>";
115 send = false;
116 }
117 
118 document.getElementById("textLog").innerHTML = errore;
119 	
120 if(send){
121 $.ajax({
122 type: "POST",
123 url: "https://192.168.1.70/getVmApi.php",
124 data: "idVm=" + idVm,
125 dataType: "html",
126       
127 success: function(response)
128 {
129 var dec = JSON.parse(response);
130 var nomeVm = dec.name;
131 createTemplate(nomeVm, idVm, pool.textNumero, pool.textNome, pool.textDescrizione);
132 },
133 error: function()
134 {
135 document.getElementById("textLog").style.cssText = "color:red;"
136 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
137 }
138 }); 
139 }
140 }

142 /**
143  * Funzione che avvia la creazione del template della macchina virtuale della quale si vuole eseguire il pool.
144  *
145  * @param {string} nomeVm - Nome della macchina virtuale.
146  * @param {int} idVm - Identificativo univoco della macchina virtuale.
147  * @param {int} textNumero - Numero di macchine virtuali del pool.
148  * @param {string} textNome - Nome del pool.
149  * @param {string} textDescrizione - Descrizione del pool.
150  */
151 function createTemplate(nomeVm, idVm, textNumero, textNome, textDescrizione) {
152 $.ajax({
153 type: "POST",
154 url: "https://192.168.1.70/createTemplate.php",
155 data: "nomeVm=" + textNome + "&idVm=" + idVm,
156 dataType: "html",
157       
158 success: function(response)
159 {
160 var log = "";
161 var dec = JSON.parse(response);
162 var stato = dec.detail;
163       	
164 if(stato == undefined) {
165 document.getElementById("submitDeploy").disabled = true;
166 logInit = "<-- Inizializzazione -->";
167 document.getElementById("textLog").style.cssText = "color:black;"
168 document.getElementById("textLog").innerHTML = logInit;
169 var createTemplateTimer = setInterval(function() { waitTemplate(idVm, dec,logInit,createTemplateTimer, textNumero, textNome, textDescrizione); }, 3000); 		
170 }
171 else {
172 document.getElementById("textLog").style.cssText = "color:red;"
173 document.getElementById("textLog").innerHTML = stato;
174 } 
175 },
176 error: function()
177 {
178 document.getElementById("textLog").style.cssText = "color:red;"
179 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
180 }
181 });
182 }

184 /**
185  * Funzione che controlla se il template è stato creato (stato "down").
186  *
187  * @param {int} idVm - Identificativo univoco della macchina virtuale.
188  * @param {Json} dec - Json contenente la risposta di oVirt.
189  * @param {string} logInit - Messaggio di log.
190  * @param {double} createTemplateTimer - Timer elaborazione.
191  * @param {int} textNumero - Numero di macchine virtuali del pool.
192  * @param {string} textNome - Nome del pool.
193  * @param {string} textDescrizione - Descrizione del pool.
194  */
195 function waitTemplate(idVm,dec,logInit,createTemplateTimer, textNumero, textNome, textDescrizione) {
196 var logCreate = "Creazione Template in corso..." + "<br>" + logInit;
197 document.getElementById("textLog").style.cssText = "color:black;"
198 document.getElementById("textLog").innerHTML = logCreate;       
199                   
200 $.ajax({
201 type: "POST",
202 url: "https://192.168.1.70/getVmApi.php",
203 data: "idVm=" + idVm,
204 dataType: "html",
205       
206 success: function(response)
207 {
208 var decVm = JSON.parse(response);
209     	   
210 if(decVm.status == "down"){
211 var logCreateSuccess = "Template creato con successo." + "<br>" + logCreate;
212 document.getElementById("textLog").style.cssText = "color:black;"
213 document.getElementById("textLog").innerHTML = logCreateSuccess;
214 clearInterval(createTemplateTimer); 
215 setTimeout(function(){ createPools(dec,logCreateSuccess, textNumero, textNome, textDescrizione); }, 10000);
216 }
217 },
218 error: function()
219 {
220 document.getElementById("textLog").style.cssText = "color:red;"
221 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
222 }
223 }); 
224 }

226 /**
227  * Funzione che avvia la creazione del pool della macchina virtuale a partire dal template creato.
228  *
229  * @param {Json} dec - Json contenente la risposta di oVirt.
230  * @param {string} logCreateSuccess - Messaggio di log.
231  * @param {int} textNumero - Numero di macchine virtuali del pool.
232  * @param {string} textNome - Nome del pool.
233  * @param {string} textDescrizione - Descrizione del pool.
234  */
235 function createPools(dec,logCreateSuccess, textNumero, textNome, textDescrizione) {
236 var idTemplate = dec.id;
237 var logCreatePools = "<-- Inizio creazione Pools..." + "<br>" + logCreateSuccess;
238 document.getElementById("textLog").style.cssText = "color:black;"
239 document.getElementById("textLog").innerHTML = logCreatePools;       
240                   
241 $.ajax({
242 type: "POST",
243 url: "https://192.168.1.70/createPools.php",
244 data: "idTemplate=" + idTemplate + "&nomeVm=" + textNome + "&numeroPools=" + textNumero + "&textDescrizione=" + textDescrizione,
245 dataType: "html",
246       
247 success: function(response)
248 {
249 setTimeout(function(){ createSuccessPools(logCreatePools); }, 30000);
250 },
251 error: function()
252 {
253 document.getElementById("textLog").style.cssText = "color:red;"
254 document.getElementById("textLog").innerHTML = "Chiamata Fallita! Riprovare!";
255 }
256 }); 
257 }
